%%%%% This package and config allows fancyboxed(rb) command %%%%%%%%%%%%%%%%%%%%

\usepackage[skins,theorems]{tcolorbox}
\tcbset{highlight math style={enhanced,
colframe=red,colback=white,arc=0pt,boxrule=1pt}}

\newcommand{\fboxed}[1]{
\tcbhighmath[fuzzy halo=1mm with blue!50!white,arc=2pt,
boxrule=0pt,frame hidden]{#1}
}

\newcommand{\ffboxed}[2]{
    \tcboxmath{#1}
}

\newcommand{\fcboxed}[2]{
    \tcbhighmath[fuzzy halo=1mm with #1,arc=2pt,boxrule=0pt,frame hidden]{#2}
}
